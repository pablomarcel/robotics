def _vec3(values: Iterable[str]) -> np.ndarray:
    v = list(map(float, values))

    if len(m) != 9:

def cmd_euler_to_matrix(args: argparse.Namespace):
    # interpret angles in radians; if --deg provided, convert
    a = np.array(list(map(float, args.angles)), dtype=float)
    if args.deg:
        a = np.deg2rad(a)
    order = args.order.upper()
    # Build by successive local rotations using core.SO3
    axes = dict(X=np.array([1, 0, 0.0]), Y=np.array([0, 1, 0.0]), Z=np.array([0, 0, 1.0]))
    R = np.eye(3)
    for ang, k in zip(a, order):
        R = SO3.from_axis_angle(ang, axes[k]).R @ R
    _print_matrix(R)
    _maybe_save_matrix(args.save, R)


def cmd_matrix_to_euler(args: argparse.Namespace):
    R = _mat9(args.matrix)
    order = args.order.upper()
    # Simple numeric extraction via search (keeps us SciPy-free for now)
    # We minimize || R_target - R(order, angles) ||_F over angles.
    def build(angles):
        axes = dict(X=np.array([1, 0, 0.0]), Y=np.array([0, 1, 0.0]), Z=np.array([0, 0, 1.0]))
        Rb = np.eye(3)
        for ang, k in zip(angles, order):
            Rb = SO3.from_axis_angle(ang, axes[k]).R @ Rb
        return Rb

    # small Gauss-Newton
    x = np.zeros(3)
    for _ in range(60):
        Rb = build(x)
        r = (Rb - R).ravel()
        J = np.zeros((9, 3))
        h = 1e-6
        for i in range(3):
            dx = x.copy(); dx[i] += h
            J[:, i] = (build(dx) - Rb).ravel() / h
        step, *_ = np.linalg.lstsq(J, -r, rcond=None)
        x += step
        if np.linalg.norm(step) < 1e-10:
            break
    ang = np.rad2deg(x) if args.deg else x
    print(f"{ang[0]} {ang[1]} {ang[2]}")

def cmd_random_so3(args: argparse.Namespace):
    n = int(args.n)
    out = []
    for _ in range(n):
        # Uniform random quaternion → SO(3)
        u1, u2, u3 = np.random.rand(3)
        q = Quaternion(
            np.sqrt(1 - u1) * np.sin(2 * np.pi * u2) * 0 +  # keep scalar-first; build properly below
            0,
            0, 0, 0
        )
        # Use Shoemake method for random quat
        q = Quaternion(
            np.sqrt(1-u1) * np.cos(2*np.pi*u2),
            np.sqrt(1-u1) * np.sin(2*np.pi*u2),
            np.sqrt(u1)   * np.cos(2*np.pi*u3),
            np.sqrt(u1)   * np.sin(2*np.pi*u3),
        )
        R = q.as_matrix()
        out.append(R.tolist())
    data = {"rotations": out}
    if args.out:
        write_json(args.out, data)
        print(f"Wrote {len(out)} rotations -> {OUT_DIR / args.out}")
    else:
        print(json.dumps(data, indent=2))

def cmd_batch(args: argparse.Namespace):
    """
    Batch runner.

    Input JSON (in orientation/in/) should be a list of jobs, e.g.:

    [
      {"op": "matrix-from-axis", "params": {"phi": 0.5, "axis": [0,0,1]}},
      {"op": "compose-axis", "params": {"phi1": 0.4, "axis1": [1,0,0], "phi2": 0.7, "axis2": [0,0,1]}},
      {"op": "from-quat", "params": {"quat": [0.92388, 0, 0.38268, 0]}}
    ]
    """
    spec_path = IN_DIR / args.infile
    jobs = json.loads(spec_path.read_text())
    results: List[dict] = []

    # Dispatch table to reuse single-op handlers
    for j in jobs:
        op = j.get("op", "")
        p = j.get("params", {})
        if op == "matrix-from-axis":
            R = AxisAngle(p["phi"], np.array(p["axis"])).as_matrix()
            results.append({"op": op, "result": R.tolist()})
        elif op == "compose-axis":
            R = (
                SO3.from_axis_angle(p["phi2"], np.array(p["axis2"])).R
                @ SO3.from_axis_angle(p["phi1"], np.array(p["axis1"])).R
            )
            results.append({"op": op, "result": R.tolist()})
        elif op == "to-quat":
            q = Quaternion.from_matrix(np.array(p["matrix"]).reshape(3, 3))
            results.append({"op": op, "result": [q.e0, q.e1, q.e2, q.e3]})
        elif op == "from-quat":
            q = Quaternion(*p["quat"])
            results.append({"op": op, "result": q.as_matrix().tolist()})
        elif op == "rodrigues-to-matrix":
            R = RodriguesVector(np.array(p["w"])).as_matrix()
            results.append({"op": op, "result": R.tolist()})
        elif op == "matrix-to-rodrigues":
            aa = AxisAngle.from_matrix(np.array(p["matrix"]).reshape(3,3))
            results.append({"op": op, "result": aa.to_rodrigues().w.tolist()})
        elif op == "euler-to-matrix":
            order = p.get("order", "ZYX").upper()
            ang = np.array(p["angles"], dtype=float)
            if p.get("deg", False):
                ang = np.deg2rad(ang)
            axes = dict(X=np.array([1,0,0.0]), Y=np.array([0,1,0.0]), Z=np.array([0,0,1.0]))
            R = np.eye(3)
            for a, k in zip(ang, order):
                R = SO3.from_axis_angle(a, axes[k]).R @ R
            results.append({"op": op, "result": R.tolist()})
        elif op == "expmap":
            R = expm_so3(np.array(p["omega"], dtype=float))
            results.append({"op": op, "result": R.tolist()})
        else:
            results.append({"op": op, "error": f"unknown op '{op}'"})

    write_json(args.outfile, {"results": results})
    print(f"Wrote batch results -> {OUT_DIR / args.outfile}")

"""
orientation.cli (OO version)
----------------------------
Command-line interface for the Orientation Kinematics toolkit, implemented
with a Command pattern so each subcommand is a class.

Design
------
- CLIContext: shared config (paths, I/O helpers) for commands.
- BaseCommand: abstract interface for commands.
- One class per verb (e.g., MatrixFromAxis, ComposeAxis, ToQuat, ...).
- OrientationCLI: orchestrates argparse, registers commands, dispatches run().

This file intentionally imports only the public OO API from core/utils/design/io.
"""

@dataclass(frozen=True)
class CLIContext:
    in_dir: Path = IN_DIR
    out_dir: Path = OUT_DIR

    @staticmethod
    def vec3(values: Iterable[str]) -> np.ndarray:
        v = list(map(float, values))
        if len(v) != 3:
            raise argparse.ArgumentTypeError("Expected 3 numbers for a 3-vector.")
        return np.asarray(v, dtype=float)

    @staticmethod
    def mat9(values: Iterable[str]) -> np.ndarray:
        m = list(map(float, values))
        if len(m) != 9:
            raise argparse.ArgumentTypeError("Expected 9 numbers for a 3x3 matrix (row-major).")
        return np.asarray(m, dtype=float).reshape(3, 3)

    def maybe_save_matrix(self, name: str | None, M: np.ndarray) -> None:
        if not name:
            return
        write_matrix_csv(name, M)
        print(f"Saved CSV -> {self.out_dir / name}")

    @staticmethod
    def print_matrix(M: np.ndarray) -> None:
        np.set_printoptions(precision=6, suppress=True)
        print(M)


# ---------------------------------------------------------------------------
# Command base

class BaseCommand(ABC):
    """Abstract CLI command. Subclasses must implement add_arguments and run."""

    name: str = ""
    help: str = ""

    @abstractmethod
    def add_arguments(self, parser: argparse.ArgumentParser) -> None: ...

    @abstractmethod
    def run(self, args: argparse.Namespace, ctx: CLIContext) -> None: ...

    # small utility for --save option reuse
    @staticmethod
    def add_save_option(p: argparse.ArgumentParser) -> None:
        p.add_argument("--save", type=str, help="Save 3x3 matrix as CSV in orientation/out.")

    name, help = "euler-to-matrix", "Euler angles → rotation matrix (axis order configurable)."

    def add_arguments(self, p: argparse.ArgumentParser) -> None:
        p.add_argument("--angles", nargs=3, required=True, help="Angles (default radians).")
        p.add_argument("--order", type=str, default="ZYX", help="Axis order, e.g. ZYX or XYZ.")
        p.add_argument("--deg", action="store_true", help="Interpret input angles as degrees.")
        self.add_save_option(p)

    def run(self, args: argparse.Namespace, ctx: CLIContext) -> None:
        ang = np.asarray(list(map(float, args.angles)), dtype=float)
        if args.deg:
            ang = np.deg2rad(ang)
        axes = {"X": np.array([1.0, 0.0, 0.0]),
                "Y": np.array([0.0, 1.0, 0.0]),
                "Z": np.array([0.0, 0.0, 1.0])}
        R = np.eye(3)
        for a, k in zip(ang, args.order.upper()):
            R = SO3.from_axis_angle(a, axes[k]).R @ R
        ctx.print_matrix(R)
        ctx.maybe_save_matrix(args.save, R)


class MatrixToEuler(BaseCommand):
    name, help = "matrix-to-euler", "Rotation matrix → Euler angles (solve for given order)."

    def add_arguments(self, p: argparse.ArgumentParser) -> None:
        p.add_argument("--matrix", nargs=9, required=True)
        p.add_argument("--order", type=str, default="ZYX")
        p.add_argument("--deg", action="store_true", help="Print angles in degrees.")

    def run(self, args: argparse.Namespace, ctx: CLIContext) -> None:
        # numerical solve: minimize ||R(angles) - R_target||_F
        Rtarget = ctx.mat9(args.matrix)
        order = args.order.upper()
        axes = {"X": np.array([1.0, 0.0, 0.0]),
                "Y": np.array([0.0, 1.0, 0.0]),
                "Z": np.array([0.0, 0.0, 1.0])}

        def build(a: Sequence[float]) -> np.ndarray:
            Rb = np.eye(3)
            for ang, k in zip(a, order):
                Rb = SO3.from_axis_angle(ang, axes[k]).R @ Rb
            return Rb

        x = np.zeros(3)
        for _ in range(60):
            Rb = build(x)
            r = (Rb - Rtarget).ravel()
            if np.linalg.norm(r) < 1e-12:
                break
            J = np.zeros((9, 3))
            eps = 1e-6
            for i in range(3):
                dx = x.copy(); dx[i] += eps
                J[:, i] = (build(dx) - Rb).ravel() / eps
            step, *_ = np.linalg.lstsq(J, -r, rcond=None)
            x += step
            if np.linalg.norm(step) < 1e-10:
                break

        ang = np.rad2deg(x) if args.deg else x
        print(f"{ang[0]} {ang[1]} {ang[2]}")

    name, help = "random-so3", "Generate N random SO(3) rotations (uniform)."

    def run(self, args: argparse.Namespace, ctx: CLIContext) -> None:

        rots = []
        for _ in range(n):
            # Shoemake method (uniform on S^3 → SO(3))
            u1, u2, u3 = np.random.rand(3)
            q = Quaternion(
                np.sqrt(1 - u1) * np.cos(2 * np.pi * u2),
                np.sqrt(1 - u1) * np.sin(2 * np.pi * u2),
                np.sqrt(u1) * np.cos(2 * np.pi * u3),
                np.sqrt(u1) * np.sin(2 * np.pi * u3),
            )
            rots.append(q.as_matrix().tolist())
        payload = {"rotations": rots}
        if args.out:
            write_json(args.out, payload)
            print(f"Wrote {n} rotations -> {ctx.out_dir / args.out}")

            print(json.dumps(payload, indent=2))

class Batch(BaseCommand):
    name, help = "batch", "Run a JSON batch of operations from orientation/in."

    def add_arguments(self, p: argparse.ArgumentParser) -> None:
        p.add_argument("--in", dest="infile", required=True, help="Input JSON in orientation/in/")
        p.add_argument("--out", dest="outfile", required=True, help="Output JSON in orientation/out/")

    def run(self, args: argparse.Namespace, ctx: CLIContext) -> None:
        jobs = json.loads((ctx.in_dir / args.infile).read_text())
        results: List[Dict] = []
        # A thin internal dispatcher that leverages our OO classes
        dispatch: Dict[str, BaseCommand] = {
            MatrixFromAxis.name: MatrixFromAxis(),
            ComposeAxis.name: ComposeAxis(),
            ToQuat.name: ToQuat(),
            FromQuat.name: FromQuat(),
            RodriguesToMatrix.name: RodriguesToMatrix(),
            MatrixToRodrigues.name: MatrixToRodrigues(),
            EulerToMatrix.name: EulerToMatrix(),
            ExpMap.name: ExpMap(),
        }
        for job in jobs:
            op = job.get("op", "")
            params = job.get("params", {})
            cmd = dispatch.get(op)
            if not cmd:
                results.append({"op": op, "error": f"unknown op '{op}'"})
                continue
            # Build a tiny Namespace from dict and run, capturing textual output/result
            ns = argparse.Namespace(**params, save=None)  # avoid writing files in batch unless specified
            try:
                # Where possible, return structured results
                if isinstance(cmd, (MatrixFromAxis, ComposeAxis, FromQuat,
                                    RodriguesToMatrix, EulerToMatrix, ExpMap)):
                    # commands that yield a matrix
                    # replicate run() but capture the matrix
                    if isinstance(cmd, MatrixFromAxis):
                        R = AxisAngle(ns.phi, CLIContext.vec3(ns.axis)).as_matrix()
                    elif isinstance(cmd, ComposeAxis):
                        R = (SO3.from_axis_angle(ns.phi2, CLIContext.vec3(ns.axis2)).R
                             @ SO3.from_axis_angle(ns.phi1, CLIContext.vec3(ns.axis1)).R)
                    elif isinstance(cmd, FromQuat):
                        R = Quaternion(*ns.quat).as_matrix()
                    elif isinstance(cmd, RodriguesToMatrix):
                        R = RodriguesVector(CLIContext.vec3(ns.w)).as_matrix()
                    elif isinstance(cmd, EulerToMatrix):
                        ang = np.asarray(ns.angles, dtype=float)
                        if getattr(ns, "deg", False):
                            ang = np.deg2rad(ang)
                        axes = {"X": np.array([1.0, 0.0, 0.0]),
                                "Y": np.array([0.0, 1.0, 0.0]),
                                "Z": np.array([0.0, 0.0, 1.0])}
                        R = np.eye(3)
                        for a, k in zip(ang, ns.order.upper()):
                            R = SO3.from_axis_angle(a, axes[k]).R @ R
                    else:  # ExpMap
                        R = expm_so3(CLIContext.vec3(ns.omega))
                    results.append({"op": op, "result": R.tolist()})
                elif isinstance(cmd, ToQuat):
                    q = Quaternion.from_matrix(np.asarray(ns.matrix, dtype=float).reshape(3, 3))
                    results.append({"op": op, "result": [q.e0, q.e1, q.e2, q.e3]})
                elif isinstance(cmd, MatrixToRodrigues):
                    aa = AxisAngle.from_matrix(np.asarray(ns.matrix, dtype=float).reshape(3, 3))
                    results.append({"op": op, "result": aa.to_rodrigues().w.tolist()})
                else:
                    cmd.run(ns, ctx)  # fallback
            except Exception as exc:  # keep batch resilient
                results.append({"op": op, "error": f"{type(exc).__name__}: {exc}"})

        write_json(args.outfile, {"results": results})
        print(f"Wrote batch results -> {ctx.out_dir / args.outfile}")


# ---------------------------------------------------------------------------
# CLI Orchestrator
# ---------------------------------------------------------------------------

class OrientationCLI:
    """Builds the argparse tree and dispatches to command classes."""

    def __init__(self, ctx: CLIContext | None = None):
        self.ctx = ctx or CLIContext()
        self.parser = argparse.ArgumentParser(
            prog="orientation",
            description="Orientation Kinematics CLI (OO)"
        )
        self.subparsers = self.parser.add_subparsers(dest="cmd", required=True)
        self._commands: Dict[str, BaseCommand] = {}
        self._register_default_commands()

    def _register(self, cmd: BaseCommand) -> None:
        sp = self.subparsers.add_parser(cmd.name, help=cmd.help)
        cmd.add_arguments(sp)
        sp.set_defaults(_cmd=cmd)
        self._commands[cmd.name] = cmd

    def _register_default_commands(self) -> None:
        for cls in (
            MatrixFromAxis, ComposeAxis, ToQuat, FromQuat,
            RodriguesToMatrix, MatrixToRodrigues,
            EulerToMatrix, MatrixToEuler,
            ExpMap, RandomSO3, Diagram, Batch
        ):
            self._register(cls())

    def run(self, argv: List[str] | None = None) -> None:
        args = self.parser.parse_args(argv)
        cmd: BaseCommand = getattr(args, "_cmd")
        cmd.run(args, self.ctx)


# ---------------------------------------------------------------------------
# Entry point
# ---------------------------------------------------------------------------

def main(argv: List[str] | None = None) -> None:
    OrientationCLI().run(argv)


if __name__ == "__main__":
    main()
