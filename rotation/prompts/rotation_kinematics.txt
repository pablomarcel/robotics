from __future__ import annotations
import math
import os
from dataclasses import dataclass
from typing import Iterable, List, Sequence, Tuple

import numpy as np
from numpy.typing import ArrayLike

try:
    from scipy.spatial.transform import Rotation as _R
except Exception as _e:  # pragma: no cover
    raise RuntimeError("SciPy is required: pip install scipy") from _e

R = _R  # alias

# ---------- basic helpers ----------

def ensure_dir(p: str) -> None:
    os.makedirs(p, exist_ok=True)


def parse_floats(csv: str) -> List[float]:
    if isinstance(csv, (list, tuple, np.ndarray)):
        return [float(x) for x in csv]
    # allow spaces and commas
    parts = [s for s in csv.replace(" ", "").split(",") if s != ""]
    return [float(x) for x in parts]


def parse_bool(x: str | bool | None, default: bool = False) -> bool:
    if isinstance(x, bool):
        return x
    if x is None:
        return default
    s = str(x).strip().lower()
    return s in {"1", "true", "t", "yes", "y"}


def hat(v: ArrayLike) -> np.ndarray:
    v = np.asarray(v, dtype=float).reshape(3)
    return np.array([[0.0, -v[2], v[1]],[v[2], 0.0, -v[0]],[-v[1], v[0], 0.0]], dtype=float)


def vee(M: ArrayLike) -> np.ndarray:
    M = np.asarray(M, dtype=float)
    return np.array([M[2,1] - M[1,2], M[0,2] - M[2,0], M[1,0] - M[0,1]])*0.5


def is_rotmat(Rm: ArrayLike, tol: float = 1e-8) -> Tuple[bool, dict]:
    M = np.asarray(Rm, dtype=float).reshape(3,3)
    I = np.eye(3)
    ortho = np.allclose(M.T @ M, I, atol=tol)
    det1 = np.isclose(np.linalg.det(M), 1.0, atol=1e-8)
    rows = [np.linalg.norm(M[i]) for i in range(3)]
    cols = [np.linalg.norm(M[:,j]) for j in range(3)]
    return ortho and det1, {
        "det": float(np.linalg.det(M)),
        "row_norms": rows,
        "col_norms": cols,
        "ortho_err": float(np.linalg.norm(M.T @ M - I)),
    }

# ---------- rotations & sequences ----------

VALID_TB = ["xyz","xzy","yxz","yzx","zxy","zyx"]            # Tait–Bryan (all different)
VALID_PROPER = ["zxz","xyx","yzy","zyz","xzx","yxy"]       # proper Euler (first==last)
VALID_ALL = VALID_TB + VALID_PROPER


def _axis_unit(ch: str) -> np.ndarray:
    if ch.lower() == 'x':
        return np.array([1.0,0.0,0.0])
    if ch.lower() == 'y':
        return np.array([0.0,1.0,0.0])
    if ch.lower() == 'z':
        return np.array([0.0,0.0,1.0])
    raise ValueError(f"bad axis: {ch}")


def make_basic(axis: str, angle: float, degrees: bool = False) -> R:
    ang = math.radians(angle) if degrees else angle
    return R.from_rotvec(_axis_unit(axis) * ang)


def compose_global(seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """Global/extrinsic: apply rotations about fixed axes in the listed order.
    R = R_ax_n * ... * R_ax_1  (SciPy multiplication is left-to-right application)"""
    seq = seq.lower()
    if seq not in VALID_ALL:
        raise ValueError(f"sequence must be one of {VALID_ALL}")
    if len(angles) != 3:
        raise ValueError("angles must be length-3")
    Rlist = [make_basic(ax, ang, degrees=degrees) for ax, ang in zip(seq, angles)]
    Rtot = Rlist[0]
    Rtot = Rlist[1] * Rtot
    Rtot = Rlist[2] * Rtot
    return Rtot


def compose_local(seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """Local/intrinsic: rotate about moving/body axes.
    Equivalent to reversed-order global rotations with reversed angles order."""
    return compose_global(seq[::-1], angles[::-1], degrees=degrees)


def decompose(Robj: R, seq: str, convention: str = 'global', degrees: bool = False) -> np.ndarray:
    """Return angles for a given sequence. For local sequences, use the reversed
    trick to match our compose_local semantics."""
    seq = seq.lower()
    if convention not in {'global','local'}:
        raise ValueError("convention must be 'global' or 'local'")
    if convention == 'global':
        return Robj.as_euler(seq, degrees=degrees)
    # local
    # R = local(seq, a1,a2,a3) == global(rev_seq, a3,a2,a1)
    # So to extract [a1,a2,a3] from R, get global angles for rev_seq, then reverse
    ang_rev = Robj.as_euler(seq[::-1], degrees=degrees)
    return ang_rev[::-1]


# ---------- omega ↔ angle-rate (generic, any sequence) ----------

def _omega_from_Rdot(Robj: R, Rdot: np.ndarray, frame: str = 'body') -> np.ndarray:
    """Compute ω from Rdot using Rdot = Ω R (space frame) or Rdot = R Ω (body frame).
    Here we adopt the common *space* convention: Rdot = Ω R where Ω^ = ω×.
    For body, Ω = R.T @ Rdot.
    """
    Rm = Robj.as_matrix()
    if frame == 'space':
        Om = Rdot @ Rm.T
    elif frame == 'body':
        Om = Rm.T @ Rdot
    else:
        raise ValueError("frame must be 'space' or 'body'")
    # ensure skew-symmetric
    Om = 0.5*(Om - Om.T)
    return vee(Om)


def build_rate_map(seq: str, angles: Sequence[float], convention: str = 'global', degrees: bool = False, frame: str = 'body') -> np.ndarray:
    """Numerically build the 3x3 matrix M(angles) such that ω = M * qdot for the given
    Euler/Cardan sequence and convention. This is general and matches 2.169–2.171,
    2.211–2.222, 2.243–2.249 depending on inputs.
    frame='body' gives ω components in body; 'space' gives in global.
    """
    eps = 1e-8
    if convention == 'global':
        R0 = compose_global(seq, angles, degrees=degrees)
    else:
        R0 = compose_local(seq, angles, degrees=degrees)
    M = np.zeros((3,3), float)
    for j in range(3):
        dq = np.zeros(3)
        dq[j] = 1.0  # 1 rad/s nominal
        dt = 1e-6
        new_angles = np.array(angles, float) + dq * dt
        if degrees:
            new_angles = np.degrees(new_angles)  # if degrees True, angles are in deg
        if convention == 'global':
            R1 = compose_global(seq, new_angles if not degrees else new_angles, degrees=degrees)
        else:
            R1 = compose_local(seq, new_angles if not degrees else new_angles, degrees=degrees)
        Rdot = (R1.as_matrix() - R0.as_matrix())/dt
        M[:, j] = _omega_from_Rdot(R0, Rdot, frame=frame)
    return M


def omega_from_rates(seq: str, angles: Sequence[float], rates: Sequence[float], convention: str = 'global', degrees: bool = False, frame: str = 'body') -> np.ndarray:
    M = build_rate_map(seq, angles, convention=convention, degrees=degrees, frame=frame)
    return M @ np.asarray(rates, float).reshape(3)


def rates_from_omega(seq: str, angles: Sequence[float], omega: Sequence[float], convention: str = 'global', degrees: bool = False, frame: str = 'body') -> np.ndarray:
    M = build_rate_map(seq, angles, convention=convention, degrees=degrees, frame=frame)
    # use pseudo-inverse to handle singularities
    return np.linalg.pinv(M) @ np.asarray(omega, float).reshape(3)


# ---------- repeat/power & alignment ----------

def rotation_power(Robj: R, m: int) -> R:
    if not isinstance(m, (int, np.integer)):
        raise ValueError("m must be integer")
    rotvec = Robj.as_rotvec()
    ang = np.linalg.norm(rotvec)
    if ang < 1e-12:
        return R.identity()
    axis = rotvec/ang
    return R.from_rotvec(axis * (float(m) * ang))


def align_x_to_vector(u: Sequence[float]) -> R:
    """Build a rotation whose body x-axis ends up along u, with body y-axis lying in the global (X,Y)-plane (≈ Eq. 2.99–2.106).
    Implementation: i' = û; choose j' by projecting global K onto plane normal i'; k' = i' × j'. Columns of R are the body axes expressed in global.
    """
    u = np.asarray(u, float)
    if np.linalg.norm(u) < 1e-12:
        raise ValueError("vector u must be nonzero")
    i = u/np.linalg.norm(u)
    K = np.array([0.0,0.0,1.0])
    j = K - (K @ i) * i
    if np.linalg.norm(j) < 1e-12:
        # If u is parallel to K, choose j along global Y
        j = np.array([0.0,1.0,0.0]) if i[2] > 0 else np.array([0.0,-1.0,0.0])
    j = j/np.linalg.norm(j)
    k = np.cross(i, j)
    # Orthonormalize once just in case
    j = np.cross(k, i)
    j = j/np.linalg.norm(j)
    k = np.cross(i, j)
    Rm = np.column_stack([i, j, k])  # columns are body axes in global
    return R.from_matrix(Rm)

    ortho = np.allclose(MtM, I, atol=tol)
    det = float(np.linalg.det(M))
    det1 = np.isclose(det, 1.0, atol=1e-8)
    rows = [float(np.linalg.norm(M[i])) for i in range(3)]
    cols = [float(np.linalg.norm(M[:, j])) for j in range(3)]
    return ortho and det1, {
        "det": det,
        "row_norms": rows,
        "col_norms": cols,
        "ortho_err": float(np.linalg.norm(MtM - I)),
    }

VALID_TB = ["xyz", "xzy", "yxz", "yzx", "zxy", "zyx"]       # Tait–Bryan
VALID_PROPER = ["zxz", "xyx", "yzy", "zyz", "xzx", "yxy"]  # Proper Euler
VALID_ALL = VALID_TB + VALID_PROPER

# ---------------------------------------------------------------------
# Composition / decomposition (global vs local, aligned with SciPy)
# ---------------------------------------------------------------------

def compose_global(seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """
    Global/extrinsic composition: fixed axes (lowercase in SciPy).
    """
    seq = seq.lower()
    if seq not in VALID_ALL:
        raise ValueError(f"sequence must be one of {VALID_ALL}")
    if len(angles) != 3:
        raise ValueError("angles must be length-3")
    return R.from_euler(seq, angles, degrees=degrees)


def compose_local(seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """
    Local/intrinsic composition: moving/body axes (UPPERCASE in SciPy).
    """
    SEQ = seq.upper()
    if seq.lower() not in VALID_ALL:
        raise ValueError(f"sequence must be one of {VALID_ALL}")
    if len(angles) != 3:
        raise ValueError("angles must be length-3")
    return R.from_euler(SEQ, angles, degrees=degrees)


def decompose(Robj: R, seq: str, convention: str = "global", degrees: bool = False) -> np.ndarray:
    """
    Extract angles for a given sequence and convention, **mirroring** compose_* above.
    """
    if convention not in {"global", "local"}:
        raise ValueError("convention must be 'global' or 'local'")
    if convention == "global":
        return Robj.as_euler(seq.lower(), degrees=degrees)
    else:
        return Robj.as_euler(seq.upper(), degrees=degrees)

# ---------------------------------------------------------------------
# ω ↔ q̇ mapping using central differences in radians
# ---------------------------------------------------------------------

def _omega_from_Rdot(Robj: R, Rdot: np.ndarray, frame: str = 'body') -> np.ndarray:
    """
    Recover ω from Rdot:
      - space frame:  Rdot = Ω_space * R  => Ω_space = Rdot * R^T
      - body  frame:  Rdot = R * Ω_body   => Ω_body  = R^T * Rdot
    """
    Rm = Robj.as_matrix()
    if frame == 'space':
        Om = Rdot @ Rm.T
    elif frame == 'body':
        Om = Rm.T @ Rdot
    else:
        raise ValueError("frame must be 'space' or 'body'")
    Om = 0.5 * (Om - Om.T)  # enforce skew
    return vee(Om)


def build_rate_map(seq: str, angles: Sequence[float],
                   convention: str = 'global', degrees: bool = False,
                   frame: str = 'body') -> np.ndarray:
    """
    Numerically build M(q) so that ω = M(q) q̇, using **central difference** in radians.
    Angles given in degrees are converted to radians internally.
    """
    seq_g = seq.lower() if convention == 'global' else seq.upper()

    # Convert angles to radians for numerics
    ang = np.asarray(angles, float)
    ang_rad = np.deg2rad(ang) if degrees else ang

    R0 = R.from_euler(seq_g, ang_rad, degrees=False)

    h = 1e-8  # radians
    M = np.zeros((3, 3), float)
    for j in range(3):
        a_plus = ang_rad.copy();  a_plus[j]  += h
        a_minus = ang_rad.copy(); a_minus[j] -= h
        R_plus = R.from_euler(seq_g, a_plus,  degrees=False)
        R_minus = R.from_euler(seq_g, a_minus, degrees=False)
        Rdot = (R_plus.as_matrix() - R_minus.as_matrix()) / (2.0 * h)
        M[:, j] = _omega_from_Rdot(R0, Rdot, frame=frame)
    return M


def omega_from_rates(seq: str, angles: Sequence[float], rates: Sequence[float],
                     convention: str = 'global', degrees: bool = False,
                     frame: str = 'body') -> np.ndarray:
    """
    Map q̇ -> ω. If degrees=True, BOTH angles and rates are interpreted in degrees/deg·s⁻¹,
    and ω is returned in deg·s⁻¹.
    """
    M = build_rate_map(seq, angles, convention=convention, degrees=degrees, frame=frame)
    qdot = np.asarray(rates, float)
    if degrees:
        qdot = np.deg2rad(qdot)
    w = M @ qdot
    return np.rad2deg(w) if degrees else w


def rates_from_omega(seq: str, angles: Sequence[float], omega: Sequence[float],
                     convention: str = 'global', degrees: bool = False,
                     frame: str = 'body') -> np.ndarray:
    """
    Map ω -> q̇ via pseudoinverse. If degrees=True, BOTH angles and ω are interpreted
    in degrees/deg·s⁻¹; returned q̇ is in deg·s⁻¹.
    """
    M = build_rate_map(seq, angles, convention=convention, degrees=degrees, frame=frame)
    w = np.asarray(omega, float)
    if degrees:
        w = np.deg2rad(w)
    qdot = np.linalg.pinv(M) @ w
    return np.rad2deg(qdot) if degrees else qdot

# ---------------------------------------------------------------------
# Repeat / power via logarithm map
# ---------------------------------------------------------------------

def rotation_power(Robj: R, m: int) -> R:
    """Return R^m for integer m using axis-angle scaling (log/exp on SO(3))."""
    if not isinstance(m, (int, np.integer)):
        raise ValueError("m must be integer")
    rotvec = Robj.as_rotvec()
    ang = float(np.linalg.norm(rotvec))
    if ang < 1e-12:
        return R.identity()
    axis = rotvec / ang
    return R.from_rotvec(axis * (float(m) * ang))

# ---------------------------------------------------------------------
# Alignment: align body x-axis with a target vector
# ---------------------------------------------------------------------

def align_x_to_vector(u: Sequence[float]) -> R:
    """
    Build a rotation whose body x-axis aligns with u, with body y-axis chosen in the
    global (X,Y)-plane when possible; columns of the returned matrix are the body
    axes expressed in the global frame (right-handed).
    """
    u = np.asarray(u, dtype=float)
    n = np.linalg.norm(u)
    if n < 1e-12:
        raise ValueError("vector u must be nonzero")
    i = u / n  # body x-axis in global
    K = np.array([0.0, 0.0, 1.0])

    # pick j by projecting global K to plane normal i
    j = K - (K @ i) * i
    if np.linalg.norm(j) < 1e-12:  # u parallel to Z; fall back to Y
        j = np.array([0.0, 1.0, 0.0]) if i[2] >= 0.0 else np.array([0.0, -1.0, 0.0])
    j = j / np.linalg.norm(j)
    k = np.cross(i, j)

    # re-orthonormalize lightly to counter numeric drift
    j = np.cross(k, i); j = j / np.linalg.norm(j)
    k = np.cross(i, j)

    Rm = np.column_stack([i, j, k])  # columns are body axes in global
    return R.from_matrix(Rm)

@pytest.mark.rot
def test_repeat_12x15deg_about_Z():
    # 12 * 15° about Z => 180° about Z
    Robj = core.build_matrix("global", "zyx", [15, 0, 0], degrees=True)  # Z=15°, Y=0, X=0
    Rm = core.repeat_rotation(Robj, 12).as_matrix()

    P = np.eye(3)  # row-vectors e1,e2,e3
    Pg = core.transform_points(Robj, P)
    assert _close(Pg, Robj.as_matrix().T, 1e-6)

@pytest.mark.rot
def test_align_x_axis_right_handed_unit_det():
    u = [0.5, -0.2, 0.84]
    Robj = core.align_body_x(u)
    M = Robj.as_matrix()
    # determinant ~ +1, orthonormal
    assert np.isclose(np.linalg.det(M), 1.0, atol=1e-8)
    assert _close(M.T @ M, np.eye(3), 1e-8)
    # x-axis aligned
    x_body_in_global = M[:, 0]
    u_hat = np.array(u) / np.linalg.norm(u)
    assert _close(x_body_in_global, u_hat, 1e-8)

def ensure_dir(p: str) -> None:
    """Create parent directories for a path if they don't exist."""
    if p == "":
        return
    os.makedirs(p, exist_ok=True)


def parse_floats(csv: str | Sequence[float]) -> List[float]:
    """Parse '1,2,3' or ['1','2','3'] into [1.0, 2.0, 3.0]."""
    if isinstance(csv, (list, tuple, np.ndarray)):
        return [float(x) for x in csv]
    parts = [s for s in str(csv).replace(" ", "").split(",") if s != ""]
    return [float(x) for x in parts]


def parse_bool(x: str | bool | None, default: bool = False) -> bool:
    if isinstance(x, bool):
        return x
    if x is None:
        return default
    s = str(x).strip().lower()
    return s in {"1", "true", "t", "yes", "y"}


def hat(v: ArrayLike) -> np.ndarray:
    """so(3) hat operator: R^3 -> skew(3)."""
    v = np.asarray(v, dtype=float).reshape(3)
    return np.array(
        [[0.0, -v[2], v[1]],
         [v[2], 0.0, -v[0]],
         [-v[1], v[0], 0.0]],
        dtype=float,
    )


def vee(M: ArrayLike) -> np.ndarray:
    """vee operator: skew(3) -> R^3."""
    M = np.asarray(M, dtype=float)
    return np.array(
        [M[2, 1] - M[1, 2], M[0, 2] - M[2, 0], M[1, 0] - M[0, 1]],
        dtype=float,
    ) * 0.5


def is_rotmat(Rm: ArrayLike, tol: float = 1e-8) -> Tuple[bool, dict]:
    """Check orthogonality and det=+1; return diagnostics."""
    M = np.asarray(Rm, dtype=float).reshape(3, 3)
    I = np.eye(3)
    MtM = M.T @ M
    ortho = np.allclose(MtM, I, atol=tol)
    det = float(np.linalg.det(M))
    det1 = np.isclose(det, 1.0, atol=1e-8)
    rows = [float(np.linalg.norm(M[i])) for i in range(3)]
    cols = [float(np.linalg.norm(M[:, j])) for j in range(3)]
    return ortho and det1, {
        "det": det,
        "row_norms": rows,
        "col_norms": cols,
        "ortho_err": float(np.linalg.norm(MtM - I)),
    }

# ---------------------------------------------------------------------
# Sequences
# ---------------------------------------------------------------------

VALID_TB = ["xyz", "xzy", "yxz", "yzx", "zxy", "zyx"]       # Tait–Bryan
VALID_PROPER = ["zxz", "xyx", "yzy", "zyz", "xzx", "yxy"]  # Proper Euler
VALID_ALL = VALID_TB + VALID_PROPER

# ---------------------------------------------------------------------
# Composition / decomposition with a single, consistent convention
# ---------------------------------------------------------------------

def compose_global(seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """
    Global/extrinsic composition: fixed axes (SciPy lower-case).
    """
    seq = seq.lower()
    if seq not in VALID_ALL:
        raise ValueError(f"sequence must be one of {VALID_ALL}")
    if len(angles) != 3:
        raise ValueError("angles must be length-3")
    return R.from_euler(seq, angles, degrees=degrees)


def compose_local(seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """
    Local/intrinsic composition: moving/body axes.
    We adopt the *textbook identity* used in rot_closedform:
        local(seq, a1,a2,a3) == global(seq[::-1], a3,a2,a1)
    i.e., reverse sequence AND angle order so each angle stays with its axis.
    """
    seq_rev = seq.lower()[::-1]
    ang_rev = list(angles)[::-1]
    return compose_global(seq_rev, ang_rev, degrees=degrees)


def decompose(Robj: R, seq: str, convention: str = "global", degrees: bool = False) -> np.ndarray:
    """
    Extract angles for a given sequence and convention, mirroring compose_* above.
    For 'local', invert the same identity:
        R = local(seq, a1,a2,a3)  <=>  R = global(seq_rev, a3,a2,a1)
    So we extract on seq_rev and then reverse the angle order back.
    """
    seq = seq.lower()
    if convention not in {"global", "local"}:
        raise ValueError("convention must be 'global' or 'local'")

    if convention == "global":
        return Robj.as_euler(seq, degrees=degrees)
    else:
        seq_rev = seq[::-1]
        a_rev = Robj.as_euler(seq_rev, degrees=degrees)
        return a_rev[::-1]

# ---------------------------------------------------------------------
# ω ↔ q̇ mapping using central differences in radians
# ---------------------------------------------------------------------

def _omega_from_Rdot(Robj: R, Rdot: np.ndarray, frame: str = 'body') -> np.ndarray:
    """
    Recover ω from Rdot:
      - space frame:  Rdot = Ω_space * R  => Ω_space = Rdot * R^T
      - body  frame:  Rdot = R * Ω_body   => Ω_body  = R^T * Rdot
    """
    Rm = Robj.as_matrix()
    if frame == 'space':
        Om = Rdot @ Rm.T
    elif frame == 'body':
        Om = Rm.T @ Rdot
    else:
        raise ValueError("frame must be 'space' or 'body'")
    Om = 0.5 * (Om - Om.T)  # enforce skew
    return vee(Om)


def build_rate_map(seq: str, angles: Sequence[float],
                   convention: str = 'global', degrees: bool = False,
                   frame: str = 'body') -> np.ndarray:
    """
    Numerically build M(q) so that ω = M(q) q̇, using **central difference** in radians.
    Angles given in degrees are converted to radians internally.

    IMPORTANT: We compose rotations via compose_global/compose_local so the
    numeric Jacobian matches the same convention used by rot_closedform.E_matrix.
    """
    # Convert to radians for numerics
    ang = np.asarray(angles, float)
    q_rad = np.deg2rad(ang) if degrees else ang

    make = compose_global if convention == 'global' else compose_local

    R0 = make(seq, q_rad, degrees=False)
    h = 1e-8  # radians
    M = np.zeros((3, 3), float)
    for j in range(3):
        qp = q_rad.copy();  qp[j] += h
        qm = q_rad.copy();  qm[j] -= h
        Rp = make(seq, qp, degrees=False)
        Rm = make(seq, qm, degrees=False)
        Rdot = (Rp.as_matrix() - Rm.as_matrix()) / (2.0 * h)
        M[:, j] = _omega_from_Rdot(R0, Rdot, frame=frame)
    return M


def omega_from_rates(seq: str, angles: Sequence[float], rates: Sequence[float],
                     convention: str = 'global', degrees: bool = False,
                     frame: str = 'body') -> np.ndarray:
    """
    Map q̇ -> ω. If degrees=True, BOTH angles and rates are interpreted in degrees/deg·s⁻¹,
    and ω is returned in deg·s⁻¹.
    """
    M = build_rate_map(seq, angles, convention=convention, degrees=degrees, frame=frame)
    qdot = np.asarray(rates, float)
    if degrees:
        qdot = np.deg2rad(qdot)
    w = M @ qdot
    return np.rad2deg(w) if degrees else w


def rates_from_omega(seq: str, angles: Sequence[float], omega: Sequence[float],
                     convention: str = 'global', degrees: bool = False,
                     frame: str = 'body') -> np.ndarray:
    """
    Map ω -> q̇ via pseudoinverse. If degrees=True, BOTH angles and ω are interpreted
    in degrees/deg·s⁻¹; returned q̇ is in deg·s⁻¹.
    """
    M = build_rate_map(seq, angles, convention=convention, degrees=degrees, frame=frame)
    w = np.asarray(omega, float)
    if degrees:
        w = np.deg2rad(w)
    qdot = np.linalg.pinv(M) @ w
    return np.rad2deg(qdot) if degrees else qdot

def rotation_power(Robj: R, m: int) -> R:
    """Return R^m for integer m using axis-angle scaling (log/exp on SO(3))."""
    if not isinstance(m, (int, np.integer)):
        raise ValueError("m must be integer")
    rotvec = Robj.as_rotvec()
    ang = float(np.linalg.norm(rotvec))
    if ang < 1e-12:
        return R.identity()
    axis = rotvec / ang
    return R.from_rotvec(axis * (float(m) * ang))

# ---------------------------------------------------------------------
# Alignment: align body x-axis with a target vector
# ---------------------------------------------------------------------

def align_x_to_vector(u: Sequence[float]) -> R:
    """
    Build a rotation whose body x-axis aligns with u, with body y-axis chosen in the
    global (X,Y)-plane when possible; columns of the returned matrix are the body
    axes expressed in the global frame (right-handed).
    """
    u = np.asarray(u, dtype=float)
    n = np.linalg.norm(u)
    if n < 1e-12:
        raise ValueError("vector u must be nonzero")
    i = u / n  # body x-axis in global
    K = np.array([0.0, 0.0, 1.0])

    # pick j by projecting global K to plane normal i
    j = K - (K @ i) * i
    if np.linalg.norm(j) < 1e-12:  # u parallel to Z; fall back to Y
        j = np.array([0.0, 1.0, 0.0]) if i[2] >= 0.0 else np.array([0.0, -1.0, 0.0])
    j = j / np.linalg.norm(j)
    k = np.cross(i, j)

    # re-orthonormalize lightly to counter numeric drift
    j = np.cross(k, i); j = j / np.linalg.norm(j)
    k = np.cross(i, j)

    Rm = np.column_stack([i, j, k])  # columns are body axes in global
    return R.from_matrix(Rm)

# =============================

from __future__ import annotations
from typing import Tuple

import sympy as sp

# -----------------------------------------------------------------------------
# Elementary active rotations (symbolic)
# -----------------------------------------------------------------------------
def _Rx(a: sp.Symbol) -> sp.Matrix:
    c, s = sp.cos(a), sp.sin(a)
    return sp.Matrix([[1, 0, 0],
                      [0, c, -s],
                      [0, s,  c]])

def _Ry(a: sp.Symbol) -> sp.Matrix:
    c, s = sp.cos(a), sp.sin(a)
    return sp.Matrix([[ c, 0, s],
                      [ 0, 1, 0],
                      [-s, 0, c]])

def _Rz(a: sp.Symbol) -> sp.Matrix:
    c, s = sp.cos(a), sp.sin(a)
    return sp.Matrix([[c, -s, 0],
                      [s,  c, 0],
                      [0,  0, 1]])

_AX = {'x': _Rx, 'y': _Ry, 'z': _Rz}


# -----------------------------------------------------------------------------
# Composition helpers
# -----------------------------------------------------------------------------
def _compose_local(seq: str, q: Tuple[sp.Symbol, sp.Symbol, sp.Symbol]) -> sp.Matrix:
    """
    Local/intrinsic composition: rotate about moving axes in the sequence order.
    If seq = a1 a2 a3 (e.g., 'zyx') and q = (a1,a2,a3), then:
        R = R_axis3(q3) * R_axis2(q2) * R_axis1(q1)
    (rightmost acts first).
    """
    s = seq.lower()
    if len(s) != 3:
        raise ValueError("sequence must have length 3")
    R1 = _AX[s[0]](q[0])
    R2 = _AX[s[1]](q[1])
    R3 = _AX[s[2]](q[2])
    return R3 * R2 * R1


def _vee(S: sp.Matrix) -> sp.Matrix:
    """vee: skew(3) -> R^3 (symbolic). Assumes S is skew-symmetric."""
    return sp.Matrix([S[2, 1], S[0, 2], S[1, 0]])


# -----------------------------------------------------------------------------
# Closed-form E(q) such that ω = E(q) q̇
# -----------------------------------------------------------------------------
def E_matrix(
    seq: str,
    convention: str = 'local',
    frame: str = 'body',
    simplify: bool = True,
) -> Tuple[sp.Matrix, Tuple[sp.Symbol, sp.Symbol, sp.Symbol]]:
    """
    Return symbolic E(q) such that ω = E(q) q̇ for a 3-angle sequence.

    Args
    ----
    seq : str
        One of the standard 3-letter Euler/Tait–Bryan sequences, e.g., 'zyx', 'zxz', 'zyz', etc.
    convention : {'local','global'}
        - 'local' (intrinsic): angles applied about the *moving/body* axes in the listed order.
        - 'global' (extrinsic): angles applied about the *fixed/space* axes in the listed order.
          Using the standard equivalence:
              global(seq, a1,a2,a3) == local(seq[::-1], a3,a2,a1)
    frame : {'body','space'}
        Frame in which the angular velocity components are expressed:
        - 'body'  → Ω_b = Rᵀ Ṙ ,  ω = vee(Ω_b)
        - 'space' → Ω_s = Ṙ Rᵀ ,  ω = vee(Ω_s)
    simplify : bool
        If True, apply SymPy simplification to the result. Set False to speed up very large expressions.

    Returns
    -------
    E : sp.Matrix(3,3)
        Symbolic E(q) with columns corresponding to [ȧ1, ȧ2, ȧ3] in the *sequence* order passed.
    (a1, a2, a3) : tuple of sp.Symbol
        The symbolic angle variables used in E.

    Notes
    -----
    - This function constructs R(q) symbolically and then computes Ω via Ṙ.
    - For 'global', we map via the intrinsic equivalent:
          R_global(seq, a1,a2,a3) ≡ R_local(seq[::-1], a3,a2,a1)
      so the returned E still has columns aligned to the *original* (a1,a2,a3) labels.
    """
    seq = seq.lower()
    if convention not in {'local', 'global'}:
        raise ValueError("convention must be 'local' or 'global'")
    if frame not in {'body', 'space'}:
        raise ValueError("frame must be 'body' or 'space'")

    # Define angle symbols in the given sequence order
    a1, a2, a3 = sp.symbols('a1 a2 a3', real=True)
    q = (a1, a2, a3)

    # Build R according to chosen convention
    if convention == 'local':
        R = _compose_local(seq, q)
        # columns in E will be [ȧ1, ȧ2, ȧ3]
        label_subs = {a1: a1, a2: a2, a3: a3}
    else:
        # global(seq, a1,a2,a3) == local(seq[::-1], a3,a2,a1)
        R = _compose_local(seq[::-1], (a3, a2, a1))
        # but we keep the labels (a1,a2,a3) for the *input* seq when differentiating
        label_subs = {a1: a1, a2: a2, a3: a3}

    Rt = R.T
    cols = []
    for ak in q:
        dR = sp.diff(R, ak)
        if frame == 'space':
            Om = dR * Rt
        else:
            Om = Rt * dR
        # Enforce skew and extract vee
        Om = (Om - Om.T) / 2
        cols.append(_vee(Om))

    E = sp.Matrix.hstack(*cols)

    if simplify:
        E = sp.simplify(E)

    return E, (a1, a2, a3)


# -----------------------------------------------------------------------------
# RPY (ZYX) convenience: sequence order vs. classic RPY order
# -----------------------------------------------------------------------------
def E_matrix_rpy_zyx_body(
    convention: str = 'local',
    simplify: bool = True,
) -> Tuple[sp.Matrix, Tuple[sp.Symbol, sp.Symbol, sp.Symbol],
           sp.Matrix, Tuple[sp.Symbol, sp.Symbol, sp.Symbol]]:
    """
    Convenience for RPY (ZYX) in the body frame, returning BOTH:
      1) E in *sequence* order [a1=ψ (about z), a2=θ (about y), a3=φ (about x)]
      2) E reindexed to classic RPY column order [φ, θ, ψ] (columns aligned to [φ̇, θ̇, ψ̇])

    Returns
    -------
    (E_seq, (a1,a2,a3), E_rpy, (φ,θ,ψ))
    """
    E_seq, (a1, a2, a3) = E_matrix('zyx', convention=convention, frame='body', simplify=simplify)
    ψ, θ, φ = a1, a2, a3
    # Reindex columns to [φ̇, θ̇, ψ̇]
    E_rpy = E_seq[:, [2, 1, 0]]
    if simplify:
        E_rpy = sp.simplify(E_rpy)
    return E_seq, (a1, a2, a3), E_rpy, (φ, θ, ψ)


# -----------------------------------------------------------------------------
# Optional extras (kept general for future growth)
# -----------------------------------------------------------------------------
def E_matrix_proper(
    seq: str,
    convention: str = 'local',
    frame: str = 'body',
    simplify: bool = True,
) -> Tuple[sp.Matrix, Tuple[sp.Symbol, sp.Symbol, sp.Symbol]]:
    """
    Convenience wrapper for proper Euler sequences (e.g., 'zxz', 'zyz').
    Equivalent to calling E_matrix directly; provided for readability.
    """
    s = seq.lower()
    if s not in {'zxz', 'zyz', 'xyx', 'xzx', 'yxy', 'yzy'}:
        raise ValueError("E_matrix_proper expects a proper Euler sequence like 'zxz' or 'zyz'")
    return E_matrix(s, convention=convention, frame=frame, simplify=simplify)

from typing import Dict, Sequence

import numpy as np

# ---------------------------------------------------------------------
# Core builders / validators
# ---------------------------------------------------------------------

def build_matrix(mode: str, seq: str, angles: Sequence[float], degrees: bool = False) -> R:
    """
    Build a scipy Rotation for a given convention ('global'|'local'), sequence, and angles.
    Angles are interpreted in degrees if degrees=True, otherwise radians.
    """
    mode = mode.lower()
    if mode not in {"global", "local"}:
        raise ValueError("mode must be 'global' or 'local'")
    if mode == 'global':
        return compose_global(seq, angles, degrees=degrees)
    else:
        return compose_local(seq, angles, degrees=degrees)


def transform_points(Robj: R, pts: Sequence[Sequence[float]]) -> np.ndarray:
    """Active transform of row-stacked points/vectors using Rotation.apply."""
    P = np.asarray(pts, float)
    return Robj.apply(P)


def passive_transform(Robj: R, vecs: Sequence[Sequence[float]]) -> np.ndarray:
    """
    Passive change of coordinates (global -> body): multiply by R^T.
    Implemented via Rotation.inv().apply on row-stacked vectors.
    """
    return Robj.inv().apply(np.asarray(vecs, float))


def check_matrix(Robj: R) -> Dict:
    """Return orthogonality diagnostics for a Rotation."""
    ok, info = is_rotmat(Robj.as_matrix())
    info["ok"] = ok
    return info


# ---------------------------------------------------------------------
# Euler / RPY: ω  <->  q̇  (fixed & body frames, any proper sequence)
# ---------------------------------------------------------------------

def _vee(S: np.ndarray) -> np.ndarray:
    """vee: so(3)->R^3 for a (nearly) skew-symmetric matrix S."""
    return np.array([S[2, 1], S[0, 2], S[1, 0]], dtype=float)


def _rotation_matrix(seq: str, q_rad: Sequence[float], convention: str = 'global') -> np.ndarray:
    """
    Build a 3x3 rotation matrix for given sequence and angles (radians).
    convention: 'global' (extrinsic) or 'local' (intrinsic).
    """
    convention = convention.lower()
    if convention not in {"global", "local"}:
        raise ValueError("convention must be 'global' or 'local'")
    if convention == 'global':
        return compose_global(seq, q_rad, degrees=False).as_matrix()
    else:
        return compose_local(seq, q_rad, degrees=False).as_matrix()


def _omega_jacobian_numeric(
    seq: str,
    q_rad: np.ndarray,
    convention: str = 'global',
    frame: str = 'space',
    eps: float = 1e-8,
) -> np.ndarray:
    """
    Build numeric Jacobian M(q) so that omega = M(q) @ qdot.
    - q_rad in radians
    - frame: 'space' (ω in global) uses Ω = Ṙ Rᵀ ; 'body' (ω in body) uses Ω = Rᵀ Ṙ
    Central difference is used for robustness and correctness.
    """
    frame = frame.lower()
    if frame not in {"space", "body"}:
        raise ValueError("frame must be 'space' or 'body'")

    R0 = _rotation_matrix(seq, q_rad, convention)
    cols = []

    for i in range(3):
        dq = np.zeros(3)
        dq[i] = eps
        Rp = _rotation_matrix(seq, q_rad + dq, convention)
        Rm = _rotation_matrix(seq, q_rad - dq, convention)
        Rdot = (Rp - Rm) / (2.0 * eps)

        if frame == 'space':
            Omega = Rdot @ R0.T
        else:  # 'body'
            Omega = R0.T @ Rdot

        # enforce skew to suppress numeric noise
        Omega = 0.5 * (Omega - Omega.T)
        cols.append(_vee(Omega))

    return np.column_stack(cols)  # 3x3


def angvel_from_rates(
    seq: str,
    angles: Sequence[float],
    rates: Sequence[float],
    convention: str = 'global',
    degrees: bool = False,
    frame: str = 'body',
) -> np.ndarray:
    """
    Map generalized angle rates q̇ -> angular velocity ω.

    Parameters
    ----------
    seq : str
        Proper Euler / Tait-Bryan sequence (e.g., 'zyx', 'zyz', 'xyz', ...).
    angles : (3,)
        Angles (deg if degrees=True, else rad).
    rates : (3,)
        Angle rates q̇ (deg/s if degrees=True, else rad/s).
    convention : {'global','local'}
        'global' for extrinsic rotations, 'local' for intrinsic rotations.
    degrees : bool
        Interpret angles and rates in degrees/deg/s when True.
    frame : {'space','body'}
        Frame of ω. 'space' returns ω in global frame; 'body' in body frame.

    Returns
    -------
    ω : (3,)
        Angular velocity (deg/s if degrees=True, else rad/s).
    """
    q = np.deg2rad(angles) if degrees else np.asarray(angles, float)
    qdot = np.deg2rad(rates) if degrees else np.asarray(rates, float)

    M = _omega_jacobian_numeric(seq, q, convention=convention, frame=frame)
    omega = M @ qdot

    return np.rad2deg(omega) if degrees else omega


def rates_from_angvel(
    seq: str,
    angles: Sequence[float],
    omega: Sequence[float],
    convention: str = 'global',
    degrees: bool = False,
    frame: str = 'body',
    rcond: float = 1e-12,
) -> np.ndarray:
    """
    Map angular velocity ω -> generalized rates q̇ using a pseudoinverse.

    Parameters
    ----------
    omega : (3,)
        Angular velocity (deg/s if degrees=True, else rad/s).
    rcond : float
        Cutoff for small singular values in the pseudoinverse; helps near singularities.

    Returns
    -------
    q̇ : (3,)
        Angle rates (deg/s if degrees=True, else rad/s).
    """
    q = np.deg2rad(angles) if degrees else np.asarray(angles, float)
    w = np.deg2rad(omega) if degrees else np.asarray(omega, float)

    M = _omega_jacobian_numeric(seq, q, convention=convention, frame=frame)
    qdot = np.linalg.pinv(M, rcond=rcond) @ w

    return np.rad2deg(qdot) if degrees else qdot
